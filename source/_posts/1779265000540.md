					<h2>一、单体测试（Unit Test）简介概要</h2>

					<p>
						单体测试（Unit Test）是针对系统中的最小可测试单位进行验证的测试方式。
					</p>

					<div class="flow">
						一个方法（Method）
						一个类（Class）
						一个模块（Module）
					</div>

					<h3>单体测试核心目标</h3>

					<table>
						<tr>
							<th>验证内容</th>
							<th>示例</th>
						</tr>

						<tr>
							<td>业务逻辑</td>
							<td>金额计算是否正确</td>
						</tr>

						<tr>
							<td>条件判断</td>
							<td>if / else 分支</td>
						</tr>

						<tr>
							<td>异常处理</td>
							<td>非法参数是否抛异常</td>
						</tr>

						<tr>
							<td>边界值</td>
							<td>null / 0 / 最大值</td>
						</tr>
					</table>

				</section>

					<h2>二、单体测试总体执行步骤</h2>

					<div class="flow">
						① 分析目标方法
						↓
						② 明确业务规则
						↓
						③ 拆分测试场景
						↓
						④ 准备测试数据
						↓
						⑤ Mock 外部依赖
						↓
						⑥ 执行测试
						↓
						⑦ 验证结果
						↓
						⑧ 检查覆盖率
						↓
						⑨ 重构与优化
					</div>

				</section>

				<!-- Step1 -->
				<section class="section" id="step1">

					<h2>① 分析目标方法</h2>

					<p>
						首先需要明确目标方法的业务职责与处理流程。
					</p>

					<pre><code>public UserDto getUser(String userId) {

    if (userId == null) {
        throw new IllegalArgumentException();
    }

    UserEntity entity = userMapper.selectById(userId);

    if (entity == null) {
        throw new BusinessException("USER_NOT_FOUND");
    }

    return convert(entity);
}</code></pre>

					<h3>业务规则分析</h3>

					<table>
						<tr>
							<th>条件</th>
							<th>结果</th>
						</tr>

						<tr>
							<td>userId = null</td>
							<td>抛 IllegalArgumentException</td>
						</tr>

						<tr>
							<td>用户不存在</td>
							<td>抛 BusinessException</td>
						</tr>

						<tr>
							<td>用户存在</td>
							<td>返回 DTO</td>
						</tr>
					</table>

				</section>

				<!-- Step2 -->
				<section class="section" id="step2">

					<h2>② 拆分测试场景</h2>

					<div class="flow">
						userId == null ?
						│
						┌──────┴──────┐
						↓ ↓
						抛异常 查询数据库
						│
						entity == null ?
						│
						┌────────┴───────┐
						↓ ↓
						USER_NOT_FOUND 返回 DTO
					</div>

					<table>
						<tr>
							<th>场景ID</th>
							<th>条件</th>
							<th>预期结果</th>
						</tr>

						<tr>
							<td>01</td>
							<td>userId = null</td>
							<td>IllegalArgumentException</td>
						</tr>

						<tr>
							<td>02</td>
							<td>用户不存在</td>
							<td>BusinessException</td>
						</tr>

						<tr>
							<td>03</td>
							<td>正常用户</td>
							<td>返回 DTO</td>
						</tr>
					</table>

				</section>

				<!-- Step3 -->
				<section class="section" id="step3">

					<h2>③ Mock 外部依赖</h2>

					<div class="highlight">
						Mock 的核心目标：
						只测试业务逻辑，不真实访问数据库/API。
					</div>

					<pre><code>@Mock
private UserMapper userMapper;

when(userMapper.selectById("1001"))
    .thenReturn(userEntity);</code></pre>

					<div class="flow">
						Service
						│
						↓
						Mock UserMapper
						│
						↓
						返回假数据
					</div>

				</section>

				<!-- Step4 -->
				<section class="section" id="step4">

					<h2>④ 构造测试数据</h2>

					<div class="bad">
						❌ 不推荐大量使用 @Sql
					</div>

					<pre><code>@Sql(...)
</code></pre>

					<div class="good">
						✅ 推荐直接构造对象
					</div>

					<pre><code>UserEntity entity = new UserEntity();

entity.setUserId("1001");
entity.setUserName("Tanaka");</code></pre>

					<h3>Fixture 模式</h3>

					<pre><code>public class UserFixture {

    public static UserEntity defaultUser() {

        UserEntity entity = new UserEntity();

        entity.setUserId("1001");
        entity.setUserName("Tanaka");

        return entity;
    }
}</code></pre>

				</section>

				<!-- Step5 -->
				<section class="section" id="step5">

					<h2>⑤ 编写测试代码</h2>

					<h3>AAA 模式</h3>

					<div class="flow">
						Arrange
						↓
						准备 Mock / 数据

						Act
						↓
						执行目标方法

						Assert
						↓
						验证结果
					</div>

					<pre><code>@ExtendWith(MockitoExtension.class)
class UserServiceTest {

    @InjectMocks
    private UserService service;

    @Mock
    private UserMapper userMapper;

    @Test
    void getUser_whenUserExists_thenReturnDto() {

        // Arrange
        UserEntity entity = UserFixture.defaultUser();

        when(userMapper.selectById("1001"))
            .thenReturn(entity);

        // Act
        UserDto result = service.getUser("1001");

        // Assert
        assertEquals("Tanaka", result.getUserName());
    }
}</code></pre>

				</section>

				<!-- Step6 -->
				<section class="section" id="step6">

					<h2>⑥ 异常测试</h2>

					<pre><code>@Test
void getUser_whenUserIdNull_thenThrowException() {

    assertThrows(
        IllegalArgumentException.class,
        () -> service.getUser(null)
    );
}</code></pre>

					<h3>验证错误码</h3>

					<pre><code>BusinessException ex =
    assertThrows(
        BusinessException.class,
        () -> service.getUser("9999")
    );

assertEquals(
    "USER_NOT_FOUND",
    ex.getCode()
);</code></pre>

				</section>

				<!-- Step7 -->
				<section class="section" id="step7">

					<h2>⑦ 验证副作用</h2>

					<div class="good">
						✅ 推荐验证真正重要的业务副作用
					</div>

					<pre><code>verify(mailService)
    .sendMail(any());</code></pre>

					<div class="bad">
						❌ 不推荐验证实现细节
					</div>

					<pre><code>verify(service, times(1))
</code></pre>

				</section>

</code></pre>

					<div class="bad">

					<h2>最佳实践总结</h2>

					<table>
						<tr>
							<th>推荐</th>
							<th>说明</th>
						</tr>

						<tr>
							<td>使用 AAA 模式</td>
							<td>结构清晰</td>
						</tr>

						<tr>
							<td>Mock 外部依赖</td>
							<td>保持独立</td>
						</tr>

						<tr>
							<td>测试异常分支</td>
							<td>非常重要</td>
						</tr>

						<tr>
							<td>使用 Fixture</td>
							<td>提高可维护性</td>
						</tr>

						<tr>
							<td>测试业务结果</td>
							<td>不要测试实现细节</td>
						</tr>

					</div>

					<div class="good">
